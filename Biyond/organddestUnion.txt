select *
from mrr.tier_a_local_charges_origin_air
where

select *
from mrr.tier_a_org_normal_2019
where

select tw."Country", tw."Rate" as "rate_2020", ei.rate as "rate_2019"
from mrr.tier_a_local_charges_origin_air tw, mrr.tier_a_org_normal_2019 ei
where tw."Country" = ei.country_code and
	  tw.charge_name = ei.charge
	  and ei.unit = 'Per Kg'

select tw."Country", tw."Rate" as "rate_2020", ei.rate as "rate_2019"
from prod.charges docch, mrr.tier_a_local_charges_origin_air tw, mrr.tier_a_org_normal_2019 ei
where tw."Country" = ei.country_code and
	  tw.charge_name = ei.charge
	  and docch."chargetype"=ei.charge
	  and 
	  and ei.unit = 'Per Kg'
