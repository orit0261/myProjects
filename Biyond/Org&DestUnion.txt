select *
from mrr.tier_a_local_charges_origin_air
where

select *
from mrr.tier_a_org_normal_2019
where


select tw."Country",tw."Rate" as "rate_2020"
from mrr.tier_a_local_charges_origin_air tw,prod.document_charges docch
where tw."Country" = docch."countryname"
and tw.charge_name= docch."chargeType"
and tw."Unit" = 'Per kg'	  
	  
select tw."Country", tw."Rate" as "rate_2020", ei.rate as "rate_2019"
from mrr.tier_a_local_charges_origin_air tw, mrr.tier_a_org_normal_2019 ei
where tw."Country" = ei.country_code and
	  tw.charge_name = ei.charge
	  and ei.unit = 'Per Kg'



update prod.document_charges docch
set "last120D_value"=	
	(select tw."Rate" as "rate_2020"
	from mrr.tier_a_local_charges_origin_air tw,prod.document_charges docch
	where tw."Country" = docch."countryname"
  	and tw.charge_name= docch."chargeType"
  	and tw."Unit" = 'Per kg')
	,
	"price_value"=	
	(select ei.rate as "rate_2019"
	from  mrr.tier_a_org_normal_2019 ei,prod.document_charges docch
		where docch."chargeType" = ei.charge
			and ei."country_code" = docch."countryname"
	 	    and ei.unit = 'Per kg')
